## 1. 项目概述

### 1.1 项目目标

开发一个实时语音翻译网站，实现：
- 用户说话 → 实时识别 → 实时翻译 → 显示结果
- 端到端延迟 ≤ 2秒
- 支持中英日韩法德西等语言

### 1.2 技术栈

| 层级 | 技术 |
|------|------|
| 前端 | Vue 3 + Vite + WebSocket |
| 后端 | FastAPI + Whisper + 多翻译API |
| 部署 | Docker + Docker Compose |

### 1.3 核心要求

1. **翻译服务可切换**：支持DeepLX、Google、百度、有道等多种API，运行时可配置切换
2. **代码质量**：每个模块都要有明确的输入输出、错误处理、日志记录
3. **可测试**：每个模块都可以独立测试

```python
# backend/app/translator/base.py

from abc import ABC, abstractmethod
from typing import Optional, Dict, Any
from dataclasses import dataclass
from enum import Enum

class TranslatorError(Exception):
    """翻译器错误基类"""
    pass

class TranslatorTimeoutError(TranslatorError):
    """翻译超时"""
    pass

class TranslatorRateLimitError(TranslatorError):
    """翻译限流"""
    pass

class TranslatorAuthError(TranslatorError):
    """认证失败"""
    pass

@dataclass
class TranslationResult:
    """翻译结果"""
    original_text: str
    translated_text: str
    source_lang: str
    target_lang: str
    provider: str  # 哪个翻译服务提供的
    confidence: Optional[float] = None  # 置信度（如果API返回）
    metadata: Optional[Dict[str, Any]] = None  # 其他元数据

class BaseTranslator(ABC):
    """翻译器基类"""
    
    # 子类需要设置的属性
    PROVIDER_NAME: str = "base"
    SUPPORTED_LANGUAGES: list = []
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化翻译器
        
        Args:
            config: 配置字典，从settings获取
        """
        self.config = config
        self._is_available: Optional[bool] = None
    
    @abstractmethod
    async def translate(
        self,
        text: str,
        source_lang: str = "auto",
        target_lang: str = "en"
    ) -> TranslationResult:
        """
        翻译文本
        
        Args:
            text: 要翻译的文本
            source_lang: 源语言（auto为自动检测）
            target_lang: 目标语言
        
        Returns:
            TranslationResult: 翻译结果
        
        Raises:
            TranslatorError: 翻译失败
        """
        pass
    
    @abstractmethod
    async def detect_language(self, text: str) -> str:
        """
        检测语言
        
        Args:
            text: 要检测的文本
        
        Returns:
            str: 语言代码
        """
        pass
    
    async def health_check(self) -> bool:
        """
        健康检查
        
        Returns:
            bool: 服务是否可用
        """
        try:
            result = await self.translate("hello", "en", "zh")
            self._is_available = True
            return True
        except Exception:
            self._is_available = False
            return False
    
    @property
    def is_available(self) -> Optional[bool]:
        """服务是否可用（基于上次健康检查）"""
        return self._is_available
    
    def is_language_supported(self, lang_code: str) -> bool:
        """检查语言是否支持"""
        if not self.SUPPORTED_LANGUAGES:
            return True  # 如果没有定义支持列表，默认支持所有
        return lang_code.lower() in [l.lower() for l in self.SUPPORTED_LANGUAGES]
```

**验收标准**：
- [ ] 基类定义完整，包含所有必要方法

- [ ] 支持健康检查

**关键代码（翻译器工厂）**：

```python
# backend/app/translator/factory.py

from typing import Dict, Type, List, Optional
from .base import BaseTranslator, TranslatorError
from .deeplx import DeepLXTranslator
from .google import GoogleTranslator
from .baidu import BaiduTranslator
from .youdao import YoudaoTranslator
from ..config import settings
from ..utils.logger import get_logger

logger = get_logger(__name__)

class TranslatorFactory:
    """翻译器工厂"""
    
    # 注册的翻译器类
    _translators: Dict[str, Type[BaseTranslator]] = {
        "deeplx": DeepLXTranslator,
        "google": GoogleTranslator,
        "baidu": BaiduTranslator,
        "youdao": YoudaoTranslator,
    }
    
    # 单例实例缓存
    _instances: Dict[str, BaseTranslator] = {}
    
    @classmethod
    def register(cls, name: str, translator_class: Type[BaseTranslator]):
        """
        注册新的翻译器
        
        Args:
            name: 翻译器名称
            translator_class: 翻译器类
        """
        cls._translators[name] = translator_class
        logger.info(f"Registered translator: {name}")
    
    @classmethod
    def create(cls, provider: str, config: Optional[Dict] = None) -> BaseTranslator:
        """
        创建翻译器实例
        
        Args:
            provider: 翻译服务提供者
            config: 配置（可选，默认使用settings）
        
        Returns:
            BaseTranslator: 翻译器实例
        
        Raises:
            ValueError: 不支持的翻译服务
        """
        if provider not in cls._translators:
            raise ValueError(
                f"Unsupported translator: {provider}. "
                f"Available: {list(cls._translators.keys())}"
            )
        
        # 使用缓存的实例（单例模式）
        if provider in cls._instances:
            return cls._instances[provider]
        
        # 创建新实例
        translator_class = cls._translators[provider]
        
        # 获取对应配置
        if config is None:
            config = cls._get_provider_config(provider)
        
        instance = translator_class(config)
        cls._instances[provider] = instance
        
        logger.info(f"Created translator instance: {provider}")
        return instance
    
    @classmethod
    def _get_provider_config(cls, provider: str) -> Dict:
        """获取翻译服务的配置"""
        config_map = {
            "deeplx": {
                "endpoint": settings.DEEPLX_ENDPOINT,
                "timeout": settings.DEEPLX_TIMEOUT,
            },
            "google": {
                "endpoint": settings.GOOGLE_TRANSLATE_ENDPOINT,
                "api_key": settings.GOOGLE_API_KEY,
            },
            "baidu": {
                "endpoint": settings.BAIDU_ENDPOINT,
                "app_id": settings.BAIDU_APP_ID,
                "secret_key": settings.BAIDU_SECRET_KEY,
            },
            "youdao": {
                "endpoint": settings.YOUDAO_ENDPOINT,
                "app_key": settings.YOUDAO_APP_KEY,
                "secret_key": settings.YOUDAO_SECRET_KEY,
            },
        }
        return config_map.get(provider, {})
    
    @classmethod
    def get_available_providers(cls) -> List[str]:
        """获取所有可用的翻译服务"""
        return list(cls._translators.keys())


class TranslatorManager:
    """翻译器管理器（支持自动切换和降级）"""
    
    def __init__(
        self,
        primary_provider: str,
        fallback_providers: Optional[List[str]] = None,
        auto_fallback: bool = True
    ):
        """
        初始化翻译器管理器
        
        Args:
            primary_provider: 主翻译服务
            fallback_providers: 备用翻译服务列表
            auto_fallback: 是否自动降级
        """
        self.primary_provider = primary_provider
        self.fallback_providers = fallback_providers or []
        self.auto_fallback = auto_fallback
        
        self._primary_translator = TranslatorFactory.create(primary_provider)
        self._fallback_translators = [
            TranslatorFactory.create(p) for p in self.fallback_providers
        ]
        
        # 当前使用的翻译器
        self._current_translator = self._primary_translator
        self._current_provider = primary_provider
        
        logger.info(
            f"TranslatorManager initialized. "
            f"Primary: {primary_provider}, "
            f"Fallbacks: {fallback_providers}"
        )
    
    async def translate(
        self,
        text: str,
        source_lang: str = "auto",
        target_lang: str = "en"
    ):
        """
        翻译文本（支持自动降级）
        
        Args:
            text: 要翻译的文本
            source_lang: 源语言
            target_lang: 目标语言
        
        Returns:
            TranslationResult: 翻译结果
        """
        # 尝试主翻译器
        try:
            result = await self._current_translator.translate(
                text, source_lang, target_lang
            )
            return result
        except TranslatorError as e:
            logger.warning(
                f"Translation failed with {self._current_provider}: {e}"
            )
            
            if not self.auto_fallback:
                raise
            
            # 尝试备用翻译器
            for translator in self._fallback_translators:
                try:
                    result = await translator.translate(
                        text, source_lang, target_lang
                    )
                    
                    # 更新当前翻译器
                    self._current_translator = translator
                    self._current_provider = translator.PROVIDER_NAME
                    
                    logger.info(
                        f"Switched to fallback translator: {self._current_provider}"
                    )
                    
                    return result
                except TranslatorError as fallback_e:
                    logger.warning(
                        f"Fallback translation failed with {translator.PROVIDER_NAME}: {fallback_e}"
                    )
                    continue
            
            # 所有翻译器都失败
            raise TranslatorError(
                f"All translators failed. Primary: {self.primary_provider}, "
                f"Fallbacks: {self.fallback_providers}"
            )
    
    def switch_provider(self, provider: str):
        """
        手动切换翻译服务
        
        Args:
            provider: 翻译服务名称
        """
        self._current_translator = TranslatorFactory.create(provider)
        self._current_provider = provider
        logger.info(f"Manually switched to translator: {provider}")
    
    @property
    def current_provider(self) -> str:
        """当前使用的翻译服务"""
        return self._current_provider


# 全局翻译器管理器实例
_translator_manager: Optional[TranslatorManager] = None

def get_translator_manager() -> TranslatorManager:
    """获取全局翻译器管理器"""
    global _translator_manager
    
    if _translator_manager is None:
        _translator_manager = TranslatorManager(
            primary_provider=settings.TRANSLATOR_PROVIDER,
            fallback_providers=settings.TRANSLATOR_FALLBACK_ORDER if settings.TRANSLATOR_FALLBACK_ENABLED else [],
            auto_fallback=settings.TRANSLATOR_FALLBACK_ENABLED
        )
    
    return _translator_manager
```

##### 1.3.1 DeepLX实现（1小时）

```python
# backend/app/translator/deeplx.py

import httpx
from typing import Optional, Dict, Any
from .base import BaseTranslator, TranslationResult, TranslatorError, TranslatorTimeoutError
from ..utils.logger import get_logger

logger = get_logger(__name__)

class DeepLXTranslator(BaseTranslator):
    """DeepLX翻译器（免费DeepL）"""
    
    PROVIDER_NAME = "deeplx"
    SUPPORTED_LANGUAGES = [
        "ZH", "EN", "JA", "KO", "FR", "DE", "ES", "PT", "IT", "RU",
        "zh", "en", "ja", "ko", "fr", "de", "es", "pt", "it", "ru"
    ]
    
    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self.endpoint = config.get("endpoint", "https://api.deeplx.org/translate")
        self.timeout = config.get("timeout", 10)
        self.client = httpx.AsyncClient(timeout=self.timeout)
        
        logger.info(f"DeepLXTranslator initialized. Endpoint: {self.endpoint}")
    
    async def translate(
        self,
        text: str,
        source_lang: str = "auto",
        target_lang: str = "ZH"
    ) -> TranslationResult:
        """翻译文本"""
        if not text or not text.strip():
            return TranslationResult(
                original_text=text,
                translated_text=text,
                source_lang=source_lang,
                target_lang=target_lang,
                provider=self.PROVIDER_NAME
            )
        
        try:
            # 标准化语言代码
            source_lang = self._normalize_lang_code(source_lang)
            target_lang = self._normalize_lang_code(target_lang)
            
            response = await self.client.post(
                self.endpoint,
                json={
                    "text": text,
                    "source_lang": source_lang.upper() if source_lang != "auto" else "auto",
                    "target_lang": target_lang.upper()
                }
            )
            response.raise_for_status()
            
            data = response.json()
            
            # DeepLX返回格式: {"code": 200, "data": "翻译结果"}
            if data.get("code") != 200:
                raise TranslatorError(f"DeepLX error: {data}")
            
            translated_text = data.get("data", text)
            detected_lang = data.get("source_lang", source_lang)
            
            logger.debug(
                f"DeepLX translation: {text[:50]}... -> {translated_text[:50]}..."
            )
            
            return TranslationResult(
                original_text=text,
                translated_text=translated_text,
                source_lang=detected_lang,
                target_lang=target_lang,
                provider=self.PROVIDER_NAME,
                metadata={"raw_response": data}
            )
```

```python
import whisper
import numpy as np
from typing import Optional, Dict, Any
from .base import BaseASR, ASRResult, ASRError
from ..config import settings
from ..utils.logger import get_logger

logger = get_logger(__name__)

class WhisperEngine(BaseASR):
    """Whisper语音识别引擎"""
    
    # Whisper支持的语言
    SUPPORTED_LANGUAGES = [
        "zh", "en", "ja", "ko", "fr", "de", "es", "pt", "it", "ru",
        "ar", "hi", "th", "vi", "nl", "pl", "tr", "uk", "vi", "sv"
    ]
    
    def __init__(self, config: Optional[Dict[str, Any]] = None):
        super().__init__(config or {})
        
        self.model_size = self.config.get("model_size", settings.WHISPER_MODEL)
        self.device = self.config.get("device", settings.WHISPER_DEVICE)
        
        # 延迟加载模型（首次使用时加载）
        self._model = None
        self._is_loaded = False
        
        logger.info(
            f"WhisperEngine initialized. "
            f"Model: {self.model_size}, Device: {self.device}"
        )
    
    def _load_model(self):
        """加载模型"""
        if self._is_loaded:
            return
        
        logger.info(f"Loading Whisper model: {self.model_size}...")
        try:
            self._model = whisper.load_model(
                self.model_size,
                device=self.device
            )
            self._is_loaded = True
            logger.info(f"Whisper model loaded successfully")
        except Exception as e:
            logger.error(f"Failed to load Whisper model: {e}")
            raise ASRError(f"Failed to load Whisper model: {e}")
    
    async def transcribe(
        self,
        audio: np.ndarray,
        language: Optional[str] = None,
        task: str = "transcribe"
    ) -> ASRResult:
        """
        转录音频
        
        Args:
            audio: 16kHz单声道PCM音频（int16或float32）
            language: 语言代码（None为自动检测）
            task: "transcribe"（转录）或"translate"（翻译为英文）
        
        Returns:
            ASRResult: 识别结果
        """
        # 确保模型已加载
        self._load_model()
        
        # 检查音频有效性
        if audio is None or len(audio) == 0:
            return ASRResult(
                text="",
                language=language or "unknown",
                confidence=0.0,
                is_final=True
            )
        
        try:
            # 转换为float32格式（Whisper期望）
            if audio.dtype == np.int16:
                audio_float = audio.astype(np.float32) / 32768.0
            else:
                audio_float = audio
            
            # 转录
            result = self._model.transcribe(
                audio_float,
                language=language,
                task=task,
                fp16=False if self.device == "cpu" else True,
                verbose=False
            )
            
            text = result.get("text", "").strip()
            detected_lang = result.get("language", language or "unknown")
            
            # 获取置信度（如果可用）
            segments = result.get("segments", [])
            avg_logprob = np.mean([s.get("avg_logprob", 0) for s in segments]) if segments else 0
            confidence = np.exp(avg_logprob)  # 转换为概率
            
            logger.debug(
                f"Whisper transcribed: {text[:50]}... "
                f"(lang: {detected_lang}, confidence: {confidence:.2f})"
            )
            
            return ASRResult(
                text=text,
                language=detected_lang,
                confidence=confidence,
                is_final=True,
                segments=segments,
                metadata={"raw_result": result}
            )
            
        except Exception as e:
            logger.error(f"Whisper transcription error: {e}")
            raise ASRError(f"Whisper transcription failed: {e}")
    
    @property
    def is_loaded(self) -> bool:
        """模型是否已加载"""
        return self._is_loaded
    
    def get_model_info(self) -> Dict[str, Any]:
        """获取模型信息"""
        return {
            "model_size": self.model_size,
            "device": self.device,
            "is_loaded": self._is_loaded,
            "supported_languages": self.SUPPORTED_LANGUAGES
        }
```

```python
from fastapi import WebSocket
from typing import Dict, Optional
from collections import defaultdict
import asyncio
import time
from ..utils.logger import get_logger

logger = get_logger(__name__)

class ConnectionManager:
    """WebSocket连接管理器"""
    
    def __init__(self):
        # 活跃连接
        self.active_connections: Dict[str, WebSocket] = {}
        
        # 每个会话的音频缓冲区
        self.audio_buffers: Dict[str, bytearray] = defaultdict(bytearray)
        
        # 每个会话的配置
        self.session_configs: Dict[str, dict] = {}
        
        # 每个会话的状态
        self.session_states: Dict[str, str] = {}  # idle, listening, processing
        
        # 统计信息
        self.stats = {
            "total_connections": 0,
            "current_connections": 0,
            "total_messages": 0,
        }
    
    async def connect(self, websocket: WebSocket, session_id: str):
        """接受新连接"""
        await websocket.accept()
        self.active_connections[session_id] = websocket
        self.session_states[session_id] = "idle"
        self.session_configs[session_id] = {
            "source_lang": "auto",
            "target_lang": "ZH"
        }
        
        self.stats["total_connections"] += 1
        self.stats["current_connections"] = len(self.active_connections)
        
        logger.info(f"WebSocket connected: {session_id}. Total: {self.stats['current_connections']}")
    
    def disconnect(self, session_id: str):
        """断开连接"""
        self.active_connections.pop(session_id, None)
        self.audio_buffers.pop(session_id, None)
        self.session_configs.pop(session_id, None)
        self.session_states.pop(session_id, None)
        
        self.stats["current_connections"] = len(self.active_connections)
        
        logger.info(f"WebSocket disconnected: {session_id}. Total: {self.stats['current_connections']}")
    
    async def send_message(self, session_id: str, message: dict):
        """发送消息"""
        websocket = self.active_connections.get(session_id)
        if websocket:
            import json
            await websocket.send_json(message)
            self.stats["total_messages"] += 1
    
    async def broadcast(self, message: dict):
        """广播消息（用于系统通知）"""
        for session_id in self.active_connections:
            await self.send_message(session_id, message)
    
    def get_stats(self) -> dict:
```
